CREATE TABLE products(
	product_id SERIAL primary key,
	name varchar(255),
	description text,
	price numeric(13,6),
	sku varchar(50),
	stock int,
	variation varchar(255),
	brand varchar(50),
	category_id int
	
);

CREATE TABLE category(
	category_id serial primary key,
	category_name varchar(255)
);
